.scrollScreenLo

 EQUB LO(+8)            \ Direction 0 = +8 (pan right, scroll left)
 EQUB LO(-8)            \ Direction 1 = -8 (pan left, scroll right)
 EQUB LO(-320)          \ Direction 2 = -320 (pan up, scroll down)
 EQUB LO(+320)          \ Direction 3 = +320 (pan down, scroll up)
