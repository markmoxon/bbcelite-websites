 PLA

 LSR A                  \
 AND #%00011110         \ So T is a number in the range 0 to 15
 LSR A

 ADC T                  \ Set A = A + T
                        \
                        \ So A is a number in the range 0 to 22

 RTS                    \ Return from the subroutine
